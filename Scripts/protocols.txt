"""
Dataset splitting protocols.

Biased
Standard
UltraStrict
"""

from sklearn.model_selection import train_test_split


def biased_split(
    df,
    features,
    target,
    test_size=0.20,
    random_state=42
):
    """
    Random stratified split.
    """

    train, test = train_test_split(

        df,

        test_size=test_size,

        random_state=random_state,

        stratify=df[target]

    )

    return (

        train[features],
        test[features],

        train[target],
        test[target]

    )


def standard_split(
    df,
    features,
    target,
    date_column,
    test_size=0.20
):
    """
    Chronological split (80/20).
    """

    data = (
        df
        .sort_values(date_column)
        .reset_index(drop=True)
    )

    cut = int(len(data) * (1 - test_size))

    train = data.iloc[:cut]

    test = data.iloc[cut:]

    return (

        train[features],
        test[features],

        train[target],
        test[target]

    )


def ultrastrict_split(
    df,
    features,
    target,
    date_column
):
    """
    Expanding walk-forward split.

    Yields one fold per year.
    """

    data = df.copy()

    data["year"] = data[date_column].dt.year

    years = sorted(data["year"].unique())

    for year in years[1:]:

        train = data[
            data["year"] < year
        ]

        test = data[
            data["year"] == year
        ]

        if len(train) == 0:
            continue

        if len(test) == 0:
            continue

        if test[target].nunique() < 2:
            continue

        yield (

            year,

            train[features],
            test[features],

            train[target],
            test[target]

        )
