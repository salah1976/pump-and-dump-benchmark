from pathlib import Path

import pandas as pd

from src.config import RAW_DATA, PROCESSED_DATA
from src.features import (
    clean_market_data,
    add_returns,
    add_volume_features,
    add_volatility,
    add_moving_averages,
    add_rsi,
    add_ema,
    add_atr,
    add_bollinger,
    add_momentum,
    add_candlestick_features,
    add_distance_features,
)
from src.labeling import build_multiexchange_labels


INPUT = RAW_DATA / "ALL_coins_daily_multi_exchange_v5.csv"
OUTPUT = PROCESSED_DATA / "multi_exchange_benchmark.csv"


df = pd.read_csv(INPUT)

df = clean_market_data(
    df,
    group_cols=["coin","exchange"],
    date_col="timestamp",
)

df = add_returns(df, ["coin","exchange"])

df = add_volume_features(df, ["coin","exchange"], windows=(7,20))

df = add_volatility(df, ["coin","exchange"], window=7)

df = add_moving_averages(df, ["coin","exchange"], windows=(7,))

df = add_rsi(df, ["coin","exchange"])

df = add_ema(df, ["coin","exchange"])

df = add_atr(df, ["coin","exchange"])

df = add_bollinger(df, ["coin","exchange"])

df = add_momentum(df, ["coin","exchange"])

df = add_candlestick_features(df)

df = add_distance_features(df, ["coin","exchange"], window=20)

df = build_multiexchange_labels(df)

df.to_csv(OUTPUT,index=False)

print(df.shape)
