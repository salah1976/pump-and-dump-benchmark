"""
Machine learning models used for pump-and-dump detection.
"""

from sklearn.ensemble import RandomForestClassifier
from lightgbm import LGBMClassifier
from xgboost import XGBClassifier


RANDOM_STATE = 42


def build_random_forest(
    n_estimators=500,
    max_depth=None,
    min_samples_leaf=1,
):

    return RandomForestClassifier(
        n_estimators=n_estimators,
        max_depth=max_depth,
        min_samples_leaf=min_samples_leaf,
        random_state=RANDOM_STATE,
        n_jobs=-1,
    )


def build_lightgbm(
    n_estimators=500,
    learning_rate=0.05,
    num_leaves=31,
):

    return LGBMClassifier(
        n_estimators=n_estimators,
        learning_rate=learning_rate,
        num_leaves=num_leaves,
        random_state=RANDOM_STATE,
        verbose=-1,
    )


def build_xgboost(
    n_estimators=500,
    learning_rate=0.05,
    max_depth=6,
):

    return XGBClassifier(
        n_estimators=n_estimators,
        learning_rate=learning_rate,
        max_depth=max_depth,
        subsample=0.8,
        colsample_bytree=0.8,
        random_state=RANDOM_STATE,
        eval_metric="logloss",
        n_jobs=-1,
    )


def build_models():
    """
    Return all models used in the experiments.
    """

    return {

        "RandomForest":
        build_random_forest(),

        "LightGBM":
        build_lightgbm(),

        "XGBoost":
        build_xgboost(),

    }
