from pathlib import Path

import pandas as pd

from src.config import RAW_DATA, PROCESSED_DATA
from src.features import (
    clean_market_data,
    add_returns,
    add_volume_features,
    add_volatility,
    add_moving_averages,
    add_rsi,
    add_momentum,
    add_candlestick_features,
    add_distance_features,
)
from src.labeling import build_yahoo_labels


INPUT = RAW_DATA / "YahooFinance_25coins_2017_2025.csv"
OUTPUT = PROCESSED_DATA / "yahoo_benchmark.csv"


df = pd.read_csv(INPUT)

df = clean_market_data(
    df,
    group_cols=["ticker"],
    date_col="date",
)

df = add_returns(df, ["ticker"], periods=(1,3,7,14,30))

df = add_volume_features(df, ["ticker"], windows=(7,30))

df = add_volatility(df, ["ticker"], window=30)

df = add_moving_averages(df, ["ticker"], windows=(7,30))

df = add_rsi(df, ["ticker"])

df = add_momentum(df, ["ticker"])

df = add_candlestick_features(df)

df = add_distance_features(df, ["ticker"], window=30)

df = build_yahoo_labels(df)

df.to_csv(OUTPUT,index=False)

print(df.shape)
